cheateb_claim_interaction = {
	category = cheateb_interaction_menu_relation
	use_diplomatic_range = no
	auto_accept = yes
	ai_will_do = 0
	
	is_shown = {
		NOT = { scope:actor = scope:recipient }
	}
	
	on_accept = {
		scope:actor = {
			if = {
				limit = { has_character_flag = cheateb_targeted }
				random_living_character = {
					limit = { has_character_flag = cheateb_target }
					save_scope_as = cheateb_targetscope
				}
			}
			else = { save_scope_as = cheateb_targetscope }
		}
		scope:recipient = {
			every_held_title = {
				scope:cheateb_targetscope = {
					make_claim_strong = prev
				}
			}
		}
	}
}

cheateb_take_alltitle_interaction = {
	category = cheateb_interaction_menu_relation
	use_diplomatic_range = no
	auto_accept = yes
	ai_will_do = 0
	
	is_shown = {
		NOT = { scope:actor = scope:recipient }
	}
	
	on_accept = {
		create_title_and_vassal_change = change
		scope:change = {
			set_title_and_vassal_change_type = conquest
		}
		scope:actor = {
			if = {
				limit = { has_character_flag = cheateb_targeted }
				random_living_character = {
					limit = { has_character_flag = cheateb_target }
					save_scope_as = cheateb_targetscope
				}
			}
			else = { save_scope_as = cheateb_targetscope }
		}
		scope:recipient = {
				every_held_title = {
					change_title_holder = {
						holder = scope:cheateb_targetscope
						change = scope:change
					}
				}
		}
		resolve_title_and_vassal_change = scope:change
	}
}

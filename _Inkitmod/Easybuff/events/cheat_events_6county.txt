cheateb.6000 = {
	type = character_event
	title = cheateb.6000.county
	desc = cheateb.null
	theme = realm
	right_portrait = {
		character = scope:recipient
		animation = personality_honorable
	}

	option = {
		name = cheateb.6000.privateguard
		trigger = { scope:recipient = scope:actor }
		scope:recipient = {
			unlock_maa = privateguard
			spawn_army = {
				levies = 0
				men_at_arms = {
					type = privateguard
					stacks = 2
				}
				location = scope:recipient.capital_province
				inheritable = yes
				uses_supply = no
				name = privateguard
				
			}
		}
		scope:actor = {
			trigger_event = cheateb.6000
		}
	}

	option = {
		name = cheateb_niverdie_modifier
		trigger = { scope:recipient = scope:actor }
		scope:recipient.county = {
			if = { 
				limit = { has_county_modifier = cheateb_niverdie_modifier }
				remove_county_modifier = cheateb_niverdie_modifier
			}
			else = {
				add_county_modifier = cheateb_niverdie_modifier
			}
		}
		scope:actor = {
			trigger_event = cheateb.6000
		}
	}

	option = {
		name = cheateb_development_modifier
		scope:recipient.county = {
			if = { 
				limit = { has_county_modifier = cheateb_development_modifier }
				remove_county_modifier = cheateb_development_modifier
			}
			else = {
				add_county_modifier = cheateb_development_modifier
			}
		}
		scope:actor = {
			trigger_event = cheateb.6000
		}
	}

	option = {
		name = cheateb.6001.countychangemode
		scope:actor = {
			trigger_event = cheateb.6001
		}
	}

	option = {
		name = cheateb.6002.spawnchar

		}
	}

	option = {
		name = cheateb.6003.switchholding

		}
	}

		name = cheateb.return

			trigger_event = cheateb.0010
